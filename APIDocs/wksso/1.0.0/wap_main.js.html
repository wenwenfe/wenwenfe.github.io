import {wapCreateLoginFactory} from "./Login.Wap";

try {
    document.domain = "sogou.com";
} catch (e) {

}

let domain = location.host;
let login = wapCreateLoginFactory(domain);

/**
 * Wap操作登录实例
 * @name WKSSO
 * @global
 * @type {module:Wap/Login~Login}
 * @see module:Wap/Login~Login
 */
window.WKSSO = login;

$(function() {
    let clickType = "tap";
    //绑定登录相关事件
    $(document).on(clickType, '#login, .login, #s_login, .s_login', function (e) {
        e.preventDefault();
        e.stopPropagation();
        let cb;
        //默认登录刷新当前页， 如果有login_noreload则不刷新
        if(!$(this).hasClass("login_noreload")){
            cb = function () {
                location.reload(false);
            }
        }
        login.doLogin(cb);
    });

    $(document).on(clickType, '#goToLogout, #logout, .logout, #s_logout,.sogou_logout', function (e) {
        e.preventDefault();
        e.stopPropagation();
        //点击目标带有href属性， 则退出成功后跳转到该url
        let cb = $(this).attr("s_url");
        login.doLoginOut(cb);
    });
});
